# 当第一个收集100个金币的人！
# 如果你死了，重生的时候只有原来67％的金币

while True:
    # 找到金币并攻击敌人
    # 使用旗子和特殊的移动策略来赢得比赛！
    flag = hero.findFlag()

        pass  # “pass”是一个占位符，它没有任何作用
